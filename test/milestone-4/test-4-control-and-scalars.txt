program ControlScalars;

var
  i, total: integer;
  ok: boolean;
  r: real;

begin
  total := 0;
  for i := 1 to 3 do
  begin
    total := total + i
  end;
  while total < 8 do
  begin
    total := total + 1
  end;
  repeat
    total := total - 1
  until total == 7;
  case total of
    6: total := 0;
    7: total := total + 3
  end;
  ok := (total == 10) and not false;
  r := total / 4;
  r := r + 0.5;
  if ok then
    writeln(total, r)
end.
